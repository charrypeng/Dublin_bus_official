56 # get lat & lng values for departure and arrival bus stops and the line chosen by google maps 

57 line,dep_lat,dep_lng,arr_lat,arr_lng,google_pred = request.GET['line'],request.GET['dep_lat'],request.GET['dep_lng'],request.GET['arr_lat'],request.GET['arr_lng'],request.GET['google_pred'] 

58 

59 ''' Because of the inconsistencies which result from using different data sources I am using latitudes and logituteds to match google maps chosen 

60 bus stops with bus stops in our data. The lats and longs wont be exactly the same so I am narrowing them down to 3 decimal places. 

61 after doing some research I decoverved that the third decimal place in a latitude or longitude signifies a difference of about 100 metres 

62 this means that by narrowing it down to three decimal places we will be using a bus stop that is in a 100 metre squared area of the stop goolge maps picked. 

63 This could be improved but I think it provides a good approximation for the time of the journey ''' 

64 

65 # narrowing google lat and longs as specified above 

66 dep_lat = dep_lat[0:6] 

67 dep_lng = dep_lng[0:6] 

68 arr_lat = arr_lat[0:6] 

69 arr_lng = arr_lng[0:6] 

70 

71 # create dataframe out of csv with all the bus routes in our data 

72 df = pd.read_csv(os.path.abspath('bus/static/bus/routes_clean.csv')) 

73 

74 # format the dataframe as necessary to compare the latitude and longitudes in our data with google maps data  

75 df['longitude'] = df['longitude'].astype('str') 

76 df['latitude'] = df['latitude'].astype('str') 

77 df['longitude'] = df['longitude'].astype('str').str[0:6] 

78 df['latitude'] = df['latitude'].astype('str').str[0:6] 

79 

80 # create a new df with only the routes containing the arrival or departure stops chosen by google maps 

81 true_df = df[(df['LINEID'] == line) & (((df['latitude'] == dep_lat) & (df['longitude'] == dep_lng)) | ((df['latitude'] == arr_lat) & (df['longitude'] == arr_lng)))] 

82 

83 # search this new df of potential routes for one where BOTH departure and arrival bus stops are present and pick the first route where this is true 

84 potential_routes = [] 

85 for key,value in true_df['ROUTEID'].value_counts().items(): 

86 if value > 1: 

87 potential_routes.append(key) 

88 

89 correct_route = 'N/A' 

90 for route in potential_routes: 

91 if len([(true_df['ROUTEID'] == potential_routes[0]) & ((true_df['latitude'] == dep_lat) & (true_df['longitude'] == dep_lng))]) > 0 and \ 

92 len(true_df[(true_df['ROUTEID'] == potential_routes[0]) & ((true_df['latitude'] == arr_lat) & (true_df['longitude'] == arr_lng))]) > 0: 

93 correct_route = route 

94 break 

95 ''' honestly this way of chosing the route could be improved. 

96 it is picking any route with the start and end destination chosen by google maps without taking into account the number of stops between them 

97 this is because I am prioritising having our app make a prediction instead of using google prediction but maybe in the future we could add in some threshold 

98 i.e if google maps says there are 8 stops between departure and arrival the route we chose must havd 8 +/- 2 stops between them '''

100 if correct_route != 'N/A': 

101 # if an appropriate model was found, load that model  

102 with open(os.path.abspath('bus/static/bus/grad_boost_models/'+str(correct_route) +'_grad_boost_model.sav'), 'rb') as f: 

103 model = pickle.load(f) 

104 

105 # get the percent of the route the journey will take 

106 total_stops_on_route = df[df['ROUTEID']==correct_route]['PROGRNUMBER'].max() 

107 dep_stop_on_route = df[(df['ROUTEID']==correct_route) & (df['latitude'] == dep_lat) & (df['longitude'] == dep_lng)]['PROGRNUMBER'].min() 

108 arr_stop_on_route = df[(df['ROUTEID']==correct_route) & (df['latitude'] == arr_lat) & (df['longitude'] == arr_lng)]['PROGRNUMBER'].max() 

109 num_stops = arr_stop_on_route - dep_stop_on_route
